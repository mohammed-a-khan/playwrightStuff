// src/framework/base-web-page.ts
import { env, envNumber } from './utils/env';

// Inside the constructor
constructor(
  page: Page, 
  context: BrowserContext,
  baseUrl?: string
) {
  this.page = page;
  this.context = context;
  this.config = ConfigManager.getInstance();
  this.baseUrl = baseUrl || this.config.get<string>('baseUrl', env('BASE_URL', 'https://example.com'));
  this.defaultTimeout = this.config.get<number>('defaultTimeout', envNumber('DEFAULT_TIMEOUT', 30000));
  this.logger = new Logger(this.constructor.name);
  
  // Set up listeners for network events
  this.setupNetworkListeners();
}
